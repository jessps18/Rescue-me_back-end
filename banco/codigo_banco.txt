CREATE DATABASE rescueme;
USE rescueme;

CREATE TABLE cursos (
    id_curso INT AUTO_INCREMENT PRIMARY KEY,
    nome_curso VARCHAR(150) NOT NULL UNIQUE
);

CREATE TABLE pessoas (
    id_pessoa INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(70) NOT NULL UNIQUE,
    senha VARCHAR(255) NOT NULL,
    numero_contato VARCHAR(20) NOT NULL,
    tipo ENUM('USUARIO', 'FUNCIONARIO') NOT NULL
);

CREATE TABLE usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    pessoa_id INT NOT NULL UNIQUE,
    registro_academico int (9) NOT NULL UNIQUE,
    curso_id INT NOT NULL,
    
    FOREIGN KEY (pessoa_id)
        REFERENCES pessoas(id_pessoa),

    FOREIGN KEY (curso_id)
        REFERENCES cursos(id_curso)
);

CREATE TABLE funcionarios (
    id_funcionario INT AUTO_INCREMENT PRIMARY KEY,
    pessoa_id INT NOT NULL UNIQUE,
    cargo VARCHAR(100),
    
    FOREIGN KEY (pessoa_id)
        REFERENCES pessoas(id_pessoa)
);

CREATE TABLE objetos (
    id_objeto INT AUTO_INCREMENT PRIMARY KEY,
    nome_objeto VARCHAR(200) NOT NULL,
    descricao TEXT NOT NULL,
    local_encontrado VARCHAR(200) NOT NULL,
    onde_deixou VARCHAR(300) NOT NULL,
    data_criacao DATETIME DEFAULT CURRENT_TIMESTAMP,
    encontrado_por INT NOT NULL,
    recuperado_por INT,
    data_recuperacao DATETIME,
    ja_recuperado BOOLEAN DEFAULT FALSE,
    
    FOREIGN KEY (encontrado_por)
        REFERENCES pessoas(id_pessoa),

    FOREIGN KEY (recuperado_por)
        REFERENCES pessoas(id_pessoa)
);

CREATE TABLE denuncias (
    id_denuncia INT AUTO_INCREMENT PRIMARY KEY,
    titulo TEXT NOT NULL,
    descricao TEXT NOT NULL,
    objeto_id INT NOT NULL,
    denunciante INT NOT NULL,
    denunciado INT NOT NULL,
    status_denuncia ENUM(
        'PENDENTE',
        'EM_ANALISE',
        'RESOLVIDA',
        'RECUSADA'
    ) DEFAULT 'PENDENTE',

    justificativa TEXT,
    analisado_por INT,
    data_criacao DATETIME DEFAULT CURRENT_TIMESTAMP,
    data_resposta DATETIME,

    FOREIGN KEY (objeto_id)
        REFERENCES objetos(id_objeto),

    FOREIGN KEY (denunciante)
        REFERENCES pessoas(id_pessoa),

    FOREIGN KEY (denunciado)
        REFERENCES pessoas(id_pessoa),

    FOREIGN KEY (analisado_por)
        REFERENCES pessoas(id_pessoa)
);
